import React, { useState } from 'react';
import { View, Text, FlatList, TouchableOpacity, StyleSheet, SafeAreaView } from 'react-native';

const studentsList = [
  { id: '1', name: 'Alice Johnson' },
  { id: '2', name: 'Bob Smith' },
  { id: '3', name: 'Charlie Brown' },
  { id: '4', name: 'Diana Prince' },
  { id: '5', name: 'Ethan Hunt' },
];

export default function App() {
  const [attendance, setAttendance] = useState({});

  const markAttendance = (id, status) => {
    setAttendance({ ...attendance, [id]: status });
  };

  const getStatus = (id) => {
    return attendance[id] || 'Not Marked';
  };

  const presentCount = Object.values(attendance).filter(s => s === 'Present').length;
  const absentCount = Object.values(attendance).filter(s => s === 'Absent').length;

  const renderItem = ({ item }) => (
    <View style={styles.studentCard}>
      <Text style={styles.name}>{item.name}</Text>
      <View style={styles.buttons}>
        <TouchableOpacity
          style={[styles.button, { backgroundColor: 'green' }]}
          onPress={() => markAttendance(item.id, 'Present')}
        >
          <Text style={styles.buttonText}>Present</Text>
        </TouchableOpacity>
        <TouchableOpacity
          style={[styles.button, { backgroundColor: 'red' }]}
          onPress={() => markAttendance(item.id, 'Absent')}
        >
          <Text style={styles.buttonText}>Absent</Text>
        </TouchableOpacity>
      </View>
      <Text>Status: {getStatus(item.id)}</Text>
    </View>
  );

  return (
    <SafeAreaView style={styles.container}>
      <Text style={styles.title}>üìã Student Attendance App</Text>
      <FlatList
        data={studentsList}
        renderItem={renderItem}
        keyExtractor={(item) => item.id}
      />
      <View style={styles.summary}>
        <Text>‚úÖ Present: {presentCount}</Text>
        <Text>‚ùå Absent: {absentCount}</Text>
      </View>
    </SafeAreaView>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    paddingTop: 40,
    paddingHorizontal: 20,
    backgroundColor: '#fff',
  },
  title: {
    fontSize: 24,
    fontWeight: 'bold',
    marginBottom: 20,
    textAlign: 'center',
  },
  studentCard: {
    marginBottom: 15,
    padding: 15,
    borderRadius: 8,
    backgroundColor: '#f2f2f2',
  },
  name: {
    fontSize: 18,
    marginBottom: 10,
  },
  buttons: {
    flexDirection: 'row',
    marginBottom: 10,
  },
  button: {
    padding: 10,
    marginRight: 10,
    borderRadius: 5,
  },
  buttonText: {
    color: '#fff',
  },
  summary: {
    marginTop: 30,
    alignItems: 'center',
  },
});
